QBO Token Refresh — src/lib/providers/qbo.ts

export async function ensureFreshToken(entity: Entity): Promise<Entity> {
  if (entity.tokenExpiresAt && entity.tokenExpiresAt > new Date()) return entity;
  const res = await fetch('https://oauth.platform.intuit.com/oauth2/v1/tokens/bearer', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/x-www-form-urlencoded',
      Authorization: `Basic ${Buffer.from(`${process.env.QBO_CLIENT_ID}:${process.env.QBO_CLIENT_SECRET}`).toString('base64')}`,
    },
    body: new URLSearchParams({ grant_type: 'refresh_token', refresh_token: entity.refreshToken }),
  });
  const data = await res.json() as { access_token: string; refresh_token: string; expires_in: number };
  return db.entity.update({
    where: { id: entity.id },
    data: {
      accessToken: data.access_token,
      refreshToken: data.refresh_token,
      tokenExpiresAt: new Date(Date.now() + data.expires_in * 1000),
    },
  });
}

Row Quota Enforcement — src/lib/pricing.ts

const ROW_LIMITS: Record<string, number | null> = {
  FREE: 100,
  STARTER: null,
  MULTI_ENTITY: null,
};

export async function checkRowQuota(workspace: Workspace, rowCount: number) {
  const limit = ROW_LIMITS[workspace.plan];
  if (limit === null) return;
  const periodAge = Date.now() - workspace.billingPeriodStart.getTime();
  if (periodAge > 30 * 24 * 60 * 60 * 1000) {
    await db.workspace.update({ where: { id: workspace.id }, data: { monthlyRowsUsed: 0, billingPeriodStart: new Date() } });
    return;
  }
  if (workspace.monthlyRowsUsed + rowCount > limit) {
    throw new QuotaError(`Plan limit: ${workspace.monthlyRowsUsed}/${limit} rows used this month`);
  }
}

const ENTITY_LIMITS: Record<string, number> = { FREE: 1, STARTER: 1, MULTI_ENTITY: 5 };
